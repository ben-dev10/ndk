## Usage

```tsx title="nav-demo.tsx"
import NavbarStartEnd from "@/components/ndk/start-end.tsx";

<NavbarStartEnd />;
```

Note that you may have to plug-in a data source (`NavLinks`) for the links map. Here is a sample data:

```tsx
export const NavLinks = [
  {
    name: "Services",
    url: "#",
    icon: <ToolCase size={16} />,
  },
  {
    name: "Features",
    url: "#",
    icon: <Component size={16} />,
  },
  {
    name: "Blocks",
    url: "/blocks",
    icon: <Building2 size={16} />,
  },
  {
    name: "Marketplace",
    url: "#",
    icon: <ShoppingBag size={16} />,
  },
];
```
